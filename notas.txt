<script>
const fluids=[
{name:"Agua",nu:1e-6,rho:1000,color:"#3b82f6"},
{name:"Aire",nu:1.5e-5,rho:1.2,color:"#a7f3d0"},
{name:"Aceite ligero",nu:5e-5,rho:850,color:"#facc15"},
{name:"Aceite pesado",nu:2e-4,rho:900,color:"#f59e0b"},
{name:"Glicerina",nu:1e-3,rho:1260,color:"#ec4899"},
{name:"Etanol",nu:1.2e-6,rho:789,color:"#22c55e"},
{name:"Mercurio",nu:1.1e-7,rho:13534,color:"#94a3b8"},
{name:"Gasolina",nu:6e-7,rho:720,color:"#fb7185"},
{name:"Leche",nu:2e-6,rho:1030,color:"#fef9c3"},
{name:"Sangre",nu:3e-6,rho:1060,color:"#ef4444"}
];
const select=document.getElementById("fluid");
fluids.forEach((f,i)=>{
let opt=document.createElement("option");
opt.value=i;opt.text=f.name;select.appendChild(opt);
});
const ctx=document.getElementById("canvas").getContext("2d");
const chartCtx=document.getElementById("chart").getContext("2d");
let history=[];
function calc(){
let Q=parseFloat(Q_input.value);
let D=parseFloat(D_input.value);
let T=parseFloat(T_input.value);
let f=fluids[select.value];
let nu=f.nu*(1+(20-T)*0.02);
let A=Math.PI*D*D/4;
let V=Q/A;
let Re=V*D/nu;
updateUI(V,Re,nu,f);
animate(f,Re);
saveHistory(Q,D,Re,V);
drawChart();
}
function updateUI(V,Re,nu,f){
document.getElementById("props").innerHTML=
"ν="+nu.toExponential(2)+" m²/s<br>ρ="+f.rho;
document.getElementById("velCard").innerHTML="Velocidad<br>"+V.toFixed(2);
document.getElementById("reCard").innerHTML="Re<br>"+Re.toFixed(0);
let reg="Laminar",cls="laminar";
if(Re>4000){reg="Turbulento";cls="turbulento"}
else if(Re>2000){reg="Transición";cls="transicion"}
document.getElementById("regCard").innerHTML=reg;
document.getElementById("regCard").className="card "+cls;
let pos=Math.min(Re/10000,1)*100;
document.getElementById("pointer").style.left=pos+"%";
}
function animate(f,Re){
ctx.clearRect(0,0,500,200);
ctx.strokeStyle=f.color;
for(let i=0;i<20;i++){
ctx.beginPath();
for(let x=0;x<500;x+=10){
let y=100+Math.sin((x+i*10)/50);
if(Re<2000) y=100+(i-10)*3;
else if(Re<4000) y+=Math.sin(x/20+i)*10;
else y+=Math.random()*20-10;
ctx.lineTo(x,y);
}
ctx.stroke();
}
}
function saveHistory(Q,D,Re,V){
history.unshift({Q,D,Re,V});
if(history.length>10) history.pop();
let html="";
history.forEach(h=>{
html+=`<tr><td>${h.Q}</td><td>${h.D}</td><td>${h.Re.toFixed(0)}</td></tr>`;
});
document.getElementById("history").innerHTML=html;
}
function drawChart(){
chartCtx.clearRect(0,0,500,200);

// Ejes
chartCtx.strokeStyle="#888";
chartCtx.beginPath();
chartCtx.moveTo(40,10);
chartCtx.lineTo(40,180);
chartCtx.lineTo(490,180);
chartCtx.stroke();

// Títulos de ejes
chartCtx.fillStyle="#fff";
chartCtx.font="12px Arial";
chartCtx.fillText("Velocidad (m/s)",5,20);
chartCtx.fillText("Reynolds",400,195);

if(history.length<2) return;

let maxRe=Math.max(...history.map(h=>h.Re));
let maxV=Math.max(...history.map(h=>h.V));

chartCtx.beginPath();
history.forEach((h,i)=>{
let x=40+(h.Re/maxRe)*440;
let y=180-(h.V/maxV)*160;
if(i===0) chartCtx.moveTo(x,y);
else chartCtx.lineTo(x,y);
});
chartCtx.strokeStyle="#ffffff";
chartCtx.stroke();
});
chartCtx.strokeStyle="#ffffff";
chartCtx.stroke();
}
const Q_input=document.getElementById("Q");
const D_input=document.getElementById("D");
const T_input=document.getElementById("T");
[Q_input,D_input,T_input,select].forEach(el=>el.oninput=calc);
calc();
</script>
